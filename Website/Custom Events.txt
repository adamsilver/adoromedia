dataUpdated = new J2.EventHandler();

DataService = {
	init: function() {
		document.getElementById("button").onclick = this.update;
	},
	update: function() {
		var ajax = new ajax();
		ajax.callback = this.handleUpdate;
		ajax.send();
	},
	handleUpdate: function(data) {
		//fire the custom event
		dataUpdated.fire(data);
	}
}
DataService.init();


DisplayLogic = {
	init: function() {
		//subscribe to custom event
		dataUpdated.addHandler(this.handleUpdate);
	},
	handleUpdate: function(data) {
		document.getElementById("displayElement").innerHTML = data.value;
	}
}
DisplayLogic.init();
